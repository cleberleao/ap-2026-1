//Escrever um algoritmo para determinar o consumo médio de um automóvel sendo fornecida a distância total percorrida pelo automóvel e o total de combustível gasto

Algoritmo <Calcular-Consumo-Carro>
Inicio

var
distanciaTotal, totalCombustivel, consumoMedio: Real

Leia ("Digite a distancia percorrida: ", distanciaTotal)
Leia ("Digite o total de combustível consumido: ", totalCombustivel)

consumoMedio <- distanciaTotal / totalCombustivel

Escreva ("Consumo do veículo médio é: " + consumoMedio)

Fim
