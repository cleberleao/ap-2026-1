//Escrever um algoritmo que leia o nome e as três notas obtidas por um aluno de 0 a 10 durante o semestre. Calcular a soma das notas e sua média (aritmética), informar o nome e sua menção aprovado (media >= 7), Reprovado (media <= 5) e Recuperação (média entre 5.1 a 6.9).

Algoritmo <Aproveitamento-Aluno>
Inicio

var
nota1, nota2, nota3, media: Real
nome: Caractere

Leia("Escreva o nome do aluno: ", nome)
Leia("Digite a nota 1: ", nota1)
Leia("Digite a nota 2: ", nota2)
Leia("Digite a nota 3: ", nota3)

media <- (nota1 + nota2 + nota3)/3

Se (media >= 7.0) Então {
	Escreva ("O Aluno: ", nome)
	Escreva ("Foi aprovado sua nota é: ", media)
	}
Senão Se (media >= 5.1 E media <= 6.9)Então{
	Escreva ("O Aluno: ", nome)
	Escreva ("Está em recuperação sua nota é: ", media)
	}
Senão {
	Escreva ("O Aluno: ", nome)
	Escreva ("Foi reprovado sua nota é: ", media)
	}
Fim
